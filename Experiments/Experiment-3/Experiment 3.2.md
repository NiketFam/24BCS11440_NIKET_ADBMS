SELECT EMP\_CITY, COUNT(\*) AS CNT

FROM EMPLOYEES

GROUP BY EMP\_CITY

ORDER BY EMP\_CITY



\-- FIND CITY THAT HAS ONLY 1 EMPLOYEE

SELECT EMP\_CITY 

FROM EMPLOYEES

GROUP BY EMP\_CITY

HAVING COUNT(EMP\_ID)=1



\--FIND CITY THAT HAS ATLEAST 2 EMPLOYEES

SELECT EMP\_CITY 

FROM EMPLOYEES

GROUP BY EMP\_CITY

HAVING COUNT(EMP\_ID)>=2



\-- FIND THE MAXIMUM SALARY OF AN EMPLOYEE IN EACH CITY

SELECT EMP\_CITY, MAX(EMP\_SALARY) AS MAX\_SAL

FROM EMPLOYEES

GROUP BY EMP\_CITY



\--FIND THE MINIMUM SALARY OF AN EMPLOYEE IN EACH CITY

SELECT EMP\_CITY, MIN(EMP\_SALARY) AS MIN\_SAL

FROM EMPLOYEES

GROUP BY EMP\_CITY



\-- AVERAGE SALARY

SELECT EMP\_CITY, AVG(EMP\_SALARY)::NUMERIC(10,2) AS MAX\_SAL

FROM EMPLOYEES

GROUP BY EMP\_CITY



\--TO FIND DISTINCT CITY

SELECT DISTINCT EMP\_CITY 

FROM EMPLOYEES



\--- TO FIND NUMBER OF DISTINCT CITY

SELECT COUNT(DISTINCT EMP\_CITY)

FROM EMPLOYEES



\-- FIND THE NUMBER OF EMPLOYEE IN EACH CITY WHOSE SALARY IS GREATER THAN>=9K

SELECT EMP\_CITY,SUM(CASE WHEN  EMP\_SALARY>=90000 THEN 1 ELSE 0 END ) AS  COUNTS

FROM EMPLOYEES

GROUP BY EMP\_CITY



\-- NUMBER BY USING COUNT

SELECT EMP\_CITY,COUNT(CASE WHEN  EMP\_SALARY>=90000 THEN 1  END ) AS  COUNTS

FROM EMPLOYEES

GROUP BY EMP\_CITY
